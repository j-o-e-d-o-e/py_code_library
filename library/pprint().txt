pprint()

"prettyprint", for dicts

- pprint.pprint() prompts dicts or lists in alphabetically or numerically correct order
- pprint.pformat() returns this same text

EXAMPLE: 
message = "a test message"

def count(message):
	count = {}
	for character in message.lower(): #lower all characters
		count.setdefault(character, 0) #if character is not found in "message", return "0"
						#otherwise there would be an error message
		count[character] = count[character] + 1
	pprint.pprint (count) #prints dicts in alphabetical order
count(message)
